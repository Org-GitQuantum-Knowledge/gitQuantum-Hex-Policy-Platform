Code Examples

Basic Policy Evaluation

import { PolicyVM, compileToBytecode } from '@hex/core';

// Define policy rules
const rules = [
  { attribute: 'role', element: 'document', perm: 'read' },
  { attribute: 'role', element: 'document', perm: 'write' }
];

// Compile to bytecode
const bytecode = compileToBytecode(rules);
const vm = new PolicyVM(bytecode);

// Evaluate a read request
const readDecision = vm.execute({
  subject: { role: 'admin' },
  object: { type: 'document' },
  action: 'read'
});
console.log(readDecision); // 'ALLOW'

// Evaluate a delete request (not in rules)
const deleteDecision = vm.execute({
  subject: { role: 'admin' },
  object: { type: 'document' },
  action: 'delete'
});
console.log(deleteDecision); // 'DENY'

Role-Based Access Control

const rbacRules = [
  { attribute: 'role', element: 'admin_panel', perm: 'access' },
  { attribute: 'role', element: 'user_profile', perm: 'edit' },
  { attribute: 'role', element: 'reports', perm: 'view' }
];

const bytecode = compileToBytecode(rbacRules);
const vm = new PolicyVM(bytecode);

function checkAccess(userRole, resource, action) {
  const decision = vm.execute({
    subject: { role: userRole },
    object: { type: resource },
    action: action
  });
  return decision === 'ALLOW';
}

console.log(checkAccess('admin', 'admin_panel', 'access'));     // true
console.log(checkAccess('user', 'admin_panel', 'access'));      // false
console.log(checkAccess('user', 'user_profile', 'edit'));       // true

Attribute-Based Access Control

const abacRules = [
  { attribute: 'department', element: 'salary_report', perm: 'view' },
  { attribute: 'clearance_level', element: 'classified_doc', perm: 'access' }
];

const bytecode = compileToBytecode(abacRules);
const vm = new PolicyVM(bytecode);

const hrUser = {
  subject: { department: 'hr', clearance_level: 'secret' },
  object: { type: 'salary_report' },
  action: 'view'
};

const decision = vm.execute(hrUser);
console.log(decision); // 'ALLOW'

Performance Benchmarking

import { BenchmarkRunner } from '@hex/benchmark';

// Create runner with specific configuration
const runner = new BenchmarkRunner({
  iterations: 50000,
  ruleCount: 15,
  warmup: true
});

// Run single benchmark
const result = runner.run();
console.log(`HEX Time: ${result.hexTime.toFixed(2)}ms`);
console.log(`OPA Time: ${result.opaTime.toFixed(2)}ms`);
console.log(`Speedup: ${result.speedup.toFixed(2)}x`);

// Run multiple benchmarks for stability analysis
const results = runner.runMultiple(5);
const avgSpeedup = results.reduce((sum, r) => sum + r.speedup, 0) / results.length;
console.log(`Average Speedup: ${avgSpeedup.toFixed(2)}x`);

Complex Policies

// Multi-tenant SaaS policy
const multiTenantRules = [
  { attribute: 'tenant_id', element: 'tenant_data', perm: 'read' },
  { attribute: 'tenant_id', element: 'tenant_data', perm: 'write' },
  { attribute: 'is_admin', element: 'system_config', perm: 'modify' },
  { attribute: 'subscription_tier', element: 'premium_features', perm: 'access' }
];

const bytecode = compileToBytecode(multiTenantRules);
const vm = new PolicyVM(bytecode);

const tenant1User = {
  subject: { 
    tenant_id: 'tenant-123',
    is_admin: false,
    subscription_tier: 'premium'
  },
  object: { type: 'tenant_data' },
  action: 'read'
};

console.log(vm.execute(tenant1User)); // 'ALLOW'
